43      @Test
44      public void generateMarkDown() throws Exception {
45          File file = new File("../../src/site/markdown");
46          Assert.assertTrue(file.exists() && file.isDirectory());
47          file = new File(file, "reportTypes.md");
48          
49          FileOutputStream fos = new FileOutputStream(file);
50          Writer out = new BufferedWriter(new OutputStreamWriter(fos, Constants.CHARSET));
51          
52          
53          
54          
55           List<ReportTypeContainer> array = new ArrayList<ReportTypeContainer>();
56          Connection configurationDBConnection = Utility.getConfigurationDBConnection();
57          PreparedStatement prepareStatement = configurationDBConnection.prepareStatement("select classname from plugins where appliesto='REPORTING'");
58          ResultSet executeQuery = prepareStatement.executeQuery();
59          while (executeQuery.next()) {
60              ReportTypeContainer x= new ReportTypeContainer();
61              x.setType(executeQuery.getString(1));
62              array.add(x);
63          }
64          executeQuery.close();
65          prepareStatement.close();
66          configurationDBConnection.close();
67          
68          out.write("### Report Types\n\n");
69          
70          out.write("This list represents the out of the box report types provided by FGSMS. It is a plugin system and this list is actually generated during the software build"
71                  + " of FGSMS. \n\n");
72          out.write("| Report Type | Applies to (Policy Type) | Description | \n");
73          out.write("| ----------- | ------------------------ | ----------- | \n");
74          
75          List<String> displayNames = new ArrayList<String>();
76          
77          Map<String,ReportGeneratorPlugin> map = new HashMap<String, ReportGeneratorPlugin>();
78          
79          for (int i=0; i < array.size(); i++){
80              String clazz = array.get(i).getType();
81              ReportGeneratorPlugin plugin = (ReportGeneratorPlugin) Class.forName(clazz).newInstance();
82              map.put(plugin.GetDisplayName(), plugin);
83              displayNames.add(plugin.GetDisplayName());
84          }
85          
86          Collections.sort(displayNames);
87          
88          
89          for (int i=0; i < displayNames.size(); i++){
90              ReportGeneratorPlugin plugin = map.get(displayNames.get(i));
91              out.write("| ");
92              out.write(plugin.GetDisplayName());
93              //out.write(" | ");
94              //out.write(plugin.getClass().getCanonicalName());
95              out.write(" | ");
96              out.write(formatList(plugin.GetAppliesTo()));
97              out.write (" | ");
98              out.write(plugin.GetHtmlFormattedHelp());
99              
100             out.write("\n");            
101         }
102         
103         out.write("\n");
104         out.close();
105         fos.close();
106     }
107 
108     private static String formatList(List<PolicyType> list) {
109         StringBuilder sb = new StringBuilder();
110         for (int i=0; i < list.size(); i++){
111             sb.append(list.get(i).name()).append("<br>");
112         }
113         sb.setLength(sb.length()-4);
114         return sb.toString();
115     }
