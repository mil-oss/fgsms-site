41  /**
42   *  The persistent storage agent Used for transactional web service
43   * logs when they cannot be transmitted and are stored on disk This will
44   * periodically ready from disk and attempt to send back a single transaction
45   * log, if successful, all remaining items are enqueued.
46   *
47   * @author AO
48   */
49  public class PersistentStorage {

51      public static void main(String[] args) throws InterruptedException {
52          System.out.println("Usage: java -jar fgsms.AgentCore.jar");
53          System.out.println("This will monitor the configured or provided directory for encrypted files written by a Java based fgsms agent that couldn't be delievered.\n"
54                  + "These messages are read and then attempted to be delivered periodically. If delivered, they are removed from disk.This only applies to \n"
55                  + "transactions agents which cover web services. Configuration settings are within this jar file.");
56          new PersistentStorage().init(args);
57      }

60      public static void start(String[] args) {
61          instance = new PersistentStorage(args);
62          t = new Thread(new PersistentAgentThreadRunner(instance));
63          t.start();
64      }

67      public static void stop(String[] args) {
68          if (t != null && t.isAlive() && instance != null) {
69              instance.running = false;
70              try {
71                  t.join();
72              } catch (InterruptedException ex) {
73                  log.log(Level.ERROR, null, ex);
74              }
75          }
76      }

93      public void init(String[] args) throws InterruptedException {
94  
95          boolean pooled = false;
96          try {
97              file = new File("dp.lck");
98              channel = new RandomAccessFile(file, "rw").getChannel();
99              lock = channel.tryLock();
100         } catch (Exception e) {
101             // already locked
102             closeLock();
103             System.out.println("Could not obtain the lock, this means that either this program is already running or something went wrong and the file dp.lck needs to be deleted.");
104             return;
105         }
106         if (lock == null) {
107             closeLock();
108             System.out.println("Could not obtain the lock, this means that either this program is already running or something went wrong and the file dp.lck needs to be deleted.");
109             running = false;
110             return;
111         }
112 
113         if (args != null && args.length == 1 && args[0].equalsIgnoreCase("noloop")) {
114             noloop = true;
115         }
116 
117 
118 
119         Runtime.getRuntime().addShutdownHook(new RunWhenShuttingDown());
120         PropertyConfigurator.configure("log4j.properties");
121         long lastRanAt = 0;
122         long interval = 5 * 60 * 1000;        //5 min
123         long lastUpdateAt = 0;
124         long updatestatusinterval = 5 * 60 * 1000;  //5 minues
125 
126         log.log(Level.INFO, "fgsms PersistentStorageAgent startup");
127         while (running) {
128             {
129 
130                 //check last time purged
131                 if (System.currentTimeMillis() - lastRanAt > interval) {
132 
133 
134                     try {
135                         //  log.log(Level.INFO, "fgsms Data Pruner is purging old records...");
136                         boolean workdone = doWork(false);
137                     } catch (ConfigurationException ex) {
138                         log.log(Level.WARN, null, ex);
139                         running = false;
140                         done = true;
141                         //unrecoverable error
142                         return; 
143 
144                     } catch (Exception ex) {
145                         try {
146                             if (sendstatus) {
147                                 StatusHelper.tryUpdateStatus(false, "urn:" + Utility.getHostName() + ":PersistentStorageAgent" + Utility.getHostName(), ex.getMessage(), pooled, PolicyType.STATUS, Constants.UNSPECIFIED, Utility.getHostName());
148                             }
149                         } catch (Exception x) {
150                             log.log(Level.WARN, "couldn't send status update", x);
151                         }
152                     }
153                     log.log(Level.INFO, "fgsms PersistentStorageAgent sleeping for " + interval + "ms");
154                     lastRanAt = System.currentTimeMillis();
155                 }
156                 //moved below so that the status is only updated if we are configured for persistent storage, thus preventing the stale data warning
157                 if (System.currentTimeMillis() - lastUpdateAt > updatestatusinterval) {
158                     try {
159                         if (sendstatus) {
160                             StatusHelper.tryUpdateStatus(true, "urn:" + Utility.getHostName() + ":PersistentStorageAgent", "OK", pooled, PolicyType.STATUS, Constants.UNSPECIFIED, Utility.getHostName());
161                         }
162                     } catch (Exception ex) {
163                         log.log(Level.WARN, "couldn't send status update", ex);
164                     }
165                     lastUpdateAt = System.currentTimeMillis();
166 
167                 }
168             }
169             if (noloop) {
170                 running = false;
171             }
172             if (running) {
173                 Thread.sleep(5000);
174             }
175 
176         }
177         done = true;
178     }

180     /**
181      * return true if successful, false means all endpoints are down, lets delay
182      * for a bit then try again
183      *
184      * @param pooled
185      * @return
186      */
187     public static boolean doWork(boolean pooled) throws ConfigurationException {
188 
189         ConfigLoader cfg = new ConfigLoader();
190 
191         String v = cfg.getOfflinestorage();
192 
193         if (cfg.behavior == null || cfg.behavior != ConfigLoader.UnavailableBehavior.HOLDPERSIST) {
194 
195             log.log(Level.ERROR, "the current config (fgsms-agent.properties) is not configured for offline storage");
196             throw new ConfigurationException("not configured for persistant storage");
197         }
198         DataPusher.EnsureFolderExists(v);
199         File f = new File(v);
200         if (!f.exists()) {
201             try {
202                 f.createNewFile();
203             } catch (IOException ex) {
204             }
205             if (!f.exists()) {
206                 log.log(Level.FATAL, "the storage directory does not exist and I can't create it. Check the OS file permissions. " + v);
207             }
208             throw new ConfigurationException("directory does not exist");
209         }
210         //load one record
211         AddMoreData CheckPersistStore = DataPusher.CheckPersistStore();
212         if (CheckPersistStore == null) {
213             return false;
214         }
215         /*URL dcsurl = Thread.currentThread().getContextClassLoader().getResource("META-INF/" + org.miloss.fgsms.common.Constants.DCS_META);
216 
217         if (dcsurl == null) {
218             dcsurl = Thread.currentThread().getContextClassLoader().getResource("/META-INF/" + org.miloss.fgsms.common.Constants.PCS_META);
219         }*/
220         DataPusher dp = new DataPusher();
221         
222         DataPusher.Init();
223         DataCollectorService dcsservice = new DataCollectorService();
224         DCS dcsport = dcsservice.getDCSPort();
225         BindingProvider bp = (BindingProvider) dcsport;
226         Map<String, Object> context = bp.getRequestContext();
227         if (cfg.mode_ == org.miloss.fgsms.common.Constants.AuthMode.UsernamePassword) {
228             context.put(BindingProvider.USERNAME_PROPERTY, cfg.username);
229             context.put(BindingProvider.PASSWORD_PROPERTY, Utility.DE(cfg.password));
230         }
231         if (Utility.stringIsNullOrEmpty(System.getProperty("javax.net.ssl.trustStore")) && !Utility.stringIsNullOrEmpty(System.getProperty("javax.net.ssl.trustStorePassword"))) {
232             context.put("javax.net.ssl.trustStorePassword", Utility.DE(cfg.getJavaxtruststorepass()));
233             context.put("javax.net.ssl.trustStore", Utility.DE(cfg.getJavaxtruststore()));
234         }
235         try {
236             if (cfg.mode_ == org.miloss.fgsms.common.Constants.AuthMode.PKI) {
237                 if (Utility.stringIsNullOrEmpty(System.getProperty("javax.net.ssl.keyStore")) && !Utility.stringIsNullOrEmpty(System.getProperty("javax.net.ssl.keyStorePassword"))) {
238                     context.put("javax.net.ssl.keyStorePassword", Utility.DE(cfg.getJavaxkeystorepass()));
239                     context.put("javax.net.ssl.keyStore", Utility.DE(cfg.getJavaxkeystore()));
240                 }
241             }
242 
243         } catch (Exception ex) {
244             log.log(Level.FATAL, "error caught when referencing (get or set) System.properties for SSL communication. Check to ensure that this is enabled in your JAAS managemer", ex);
245         }
246         //ApacheCxfSslHelper.DoCXF(cfg.pcsport, cfg);
247 
248         boolean success = false;
249         for (int i = 0; i < cfg.DCS_URLS.size(); i++) {
250             context.put(BindingProvider.ENDPOINT_ADDRESS_PROPERTY, cfg.DCS_URLS.get(i));//
251             try {
252                 dcsport.addMoreData(CheckPersistStore.getReq());
253                 success = true;
254                 break;
255             } catch (Exception ex) {
256                 log.log(Level.WARN, "couldn't transmit the DCS message to " + cfg.DCS_URLS.get(i));
257             }
258         }
259         if (!success) {
260             DataPusher.StorePersist(CheckPersistStore);
261             return false;
262         } else {
263             CheckPersistStore = DataPusher.CheckPersistStore();
264             while (CheckPersistStore != null) {
265                 for (int i = 0; i < CheckPersistStore.getReq().size(); i++) {
266                     MessageProcessor.getSingletonObject().processPreppedMessage(CheckPersistStore.getReq().get(i));
267                 }
268                 CheckPersistStore = DataPusher.CheckPersistStore();
269             }
270             return true;
271         }
272         //try to send here
273         //if successful, delete that file from disk
274         //, read everything and add to the queue
275         //else wait and try again
276     }

278     private void closeLock() {
279         try {
280             lock.release();
281         } catch (Exception e) {
282         }
283         try {
284             channel.close();
285         } catch (Exception e) {
286         }
287     }

298     public class RunWhenShuttingDown extends Thread {
299 
300         public void run() {
301             System.out.println("Control-C caught. Shutting down...");
302             running = false;
303             while (!done) {
304                 try {
305                     Thread.sleep(1000);
306                 } catch (InterruptedException ex) {
307                 }
308             }
309             closeLock();
310             deleteFile();
311             if (!noloop) {
312                 try {
313                     if (sendstatus) {
314                         StatusHelper.tryUpdateStatus(false, "urn:" + Utility.getHostName() + ":PersistentStorageAgent", "Stopped", false, PolicyType.STATUS, Constants.UNSPECIFIED, Utility.getHostName());
315                     }
316                 } catch (Exception ex) {
317                     log.log(Level.WARN, "couldn't send status update", ex);
318                 }
319 
320             }
321         }
322     }
