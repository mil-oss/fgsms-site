327     ReportDefinition createjob2(AutomatedReportingServiceImpl instance, String owner) throws Exception {
328         AddOrUpdateScheduledReportRequestMsg request = new AddOrUpdateScheduledReportRequestMsg();
329         request.setClassification(new SecurityWrapper());
330         ReportDefinition rd = new ReportDefinition();
331         request.getJobs().add(rd);
332         rd.setExportCSVDataRequestMsg(new ExportCSVDataRequestMsg());
333         rd.getExportCSVDataRequestMsg().setClassification(new SecurityWrapper());
334         rd.getExportCSVDataRequestMsg().getURLs().add(url);
335         rd.setOwner(owner);
336         rd.getExportCSVDataRequestMsg().setExportType(ExportRecordsEnum.TRANSACTIONS);
337         rd.getExportCSVDataRequestMsg().setRange(new TimeRangeDiff());
338         rd.getExportCSVDataRequestMsg().getRange().setStart(DatatypeFactory.newInstance().newDuration(500000));
339         rd.getExportCSVDataRequestMsg().getRange().setEnd(DatatypeFactory.newInstance().newDuration(300000));
340         //   rd.setExportDataRequestMsg(new ExportDataRequestMsg());
341         rd.setEnabled(true);
342         rd.setSchedule(new ScheduleDefinition());
343         DailySchedule ds = new DailySchedule();
344         ds.setReoccurs(BigInteger.valueOf(1));
345         GregorianCalendar gcal = new GregorianCalendar();
346         gcal.setTimeInMillis(System.currentTimeMillis());
347 
348         ds.setStartingAt((gcal));
349 
350         rd.getSchedule().getTriggers().add(ds);
351 
352         try {
353             AddOrUpdateScheduledReportResponseMsg result = instance.addOrUpdateScheduledReport(request);
354             return result.getJobs().get(0);
355 
356         } catch (Exception ex) {
357             RemoveServicePolicy(url);
358             ex.printStackTrace();
359             fail("unexpected fault.");
360         }
361         return null;
362     }
