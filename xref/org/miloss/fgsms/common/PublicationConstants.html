47   public static boolean isTmodelFromfgsms(String tmodelKey)
48   {
49       if (tmodelKey==null)
50           return false;
51       String [] keys=getAllTmodelKeys();
52       for (int i=0; i < keys.length; i++)
53       {
54           if (keys[i].equalsIgnoreCase(tmodelKey))
55               return true;
56           
57       }
58       return false;
59   }
