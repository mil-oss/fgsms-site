20  package org.miloss.fgsms.agentcore;
21  
22  import org.miloss.fgsms.agentcore.mp.DefaultMessageProcessor;
23  import java.lang.management.ManagementFactory;
24  import java.util.Properties;
25  import javax.management.MBeanServer;
26  import javax.management.ObjectName;
27  import org.apache.log4j.Level;
28  import org.miloss.fgsms.common.Logger;
29  
30  /**
31   * The Message Processor is a factory/single pattern class that provides access
32   * to a singular instance of the IMessageProcessor.
33   *
34   * @see DefaultMessageProcessor
35   *
36   * @author AO
37   */
38  public class MessageProcessor {
39  
40      private final static Logger log = Logger.getLogger(org.miloss.fgsms.common.Constants.LoggerName);
41  
42      private static IMessageProcessor singletonObject;
43  
44      public static void setSingletonObject(Object object) {
45          singletonObject = (IMessageProcessor) object;
46      }
47  
48      /**
49       * gets a reference to the singleton class. be sure to call this before any
50       * other static methods
51       *
52       * @return
53       */
54      public static synchronized IMessageProcessor getSingletonObject() {
55          if (singletonObject == null) {
56              try {
57                  Properties p = ConfigLoader.loadProperties();
58                  if (p.containsKey(ConfigLoader.PROP_MESSAGE_PROCESSOR_IMPL)) {
59                      String impl = p.getProperty(ConfigLoader.PROP_MESSAGE_PROCESSOR_IMPL);
60                      IMessageProcessor mp = (IMessageProcessor) Class.forName(impl).newInstance();
61                      singletonObject = mp;
62                  }
63              } catch (ConfigurationException ex) {
64                  log.log(Level.WARN, "failed to load custom message processor implementation, falling back to default implementation", ex);
65              } catch (NullPointerException ex) {
66                  log.log(Level.WARN, "failed to load custom message processor implementation, falling back to default implementation", ex);
67              } catch (ClassNotFoundException ex) {
68                  log.log(Level.WARN, "failed to load custom message processor implementation, falling back to default implementation", ex);
69              } catch (InstantiationException ex) {
70                  log.log(Level.WARN, "failed to load custom message processor implementation, falling back to default implementation", ex);
71              } catch (IllegalAccessException ex) {
72                  log.log(Level.WARN, "failed to load custom message processor implementation, falling back to default implementation", ex);
73              } catch (Throwable t) {
74                  log.log(Level.WARN, "failed to load custom message processor implementation, falling back to default implementation", t);
75              }
76  
77              if (singletonObject == null) {
78                  singletonObject = new DefaultMessageProcessor();
79              }
80              try {
81                  MessageProcessorAdapter cache = new MessageProcessorAdapter();
82                  MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
83                  ObjectName name = new ObjectName("org.miloss.fgsms.MessageProcessor:type=MessageProcessorAdapterMBean");
84                  mbs.registerMBean(cache, name);
85              } catch (javax.management.InstanceAlreadyExistsException x) {
86  
87              } catch (Exception ex) {
88                  log.log(Level.WARN, "unable to register MessageProcessor mbean", ex);
89              }
90          }
91          return singletonObject;
92      }
93  
94  }
