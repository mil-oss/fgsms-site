37      /**
38       * {@inheritDoc}
39       */
40      @Override
41      public String GetDisplayName() {
42          return "Throughput By Service";
43      }
44  
45      /**
46       * {@inheritDoc}
47       */
48      @Override
49      public String GetHtmlFormattedHelp() {
50          return super.GetHtmlFormattedHelp() + "This represents the average invocation rate over the given period of time. The accuracy of the data"
51                  + " is limited to the time range this report was generated from. For the chart, X axis is service, Y axis is throughput (bytes total of all messages/time). "
52                  + "This data does not necessarily represent bytes on the wire and may or may not include transport overhead, headers, or packetization.";
53      }
54  
55      /**
56       * {@inheritDoc}
57       */
58      @Override
59      public void generateReport(OutputStreamWriter data, List<String> urls, String path, List<String> files, TimeRange range, String currentuser, SecurityWrapper classification, WebServiceContext ctx) throws IOException {
60  
61          Connection con = Utility.getPerformanceDBConnection();
62          try {
63              PreparedStatement cmd = null;
64              ResultSet rs = null;
65              DefaultCategoryDataset set = new DefaultCategoryDataset();
66              JFreeChart chart = null;
67              data.append("<hr /><h2>").append(GetDisplayName()).append("</h2>");
68              data.append(GetHtmlFormattedHelp() + "<br />");
69              data.append("<table  class=\"table table-hover\"><tr><th>URL</th><th>Invocations</th><th>Msg/Day</th><th>Msg/Hour</th><th>Msg/Min</th><th>Msg/Sec</th></tr>");
70              for (int i = 0; i < urls.size(); i++) {
71                  if (!isPolicyTypeOf(urls.get(i), PolicyType.TRANSACTIONAL)) {
72                      continue;
73                  }
74                  //https://github.com/mil-oss/fgsms/issues/112
75                  if (!UserIdentityUtil.hasReadAccess(currentuser, "getReport", urls.get(i), classification, ctx)) {
76                      continue;
77                  }
78                  String url = Utility.encodeHTML(BaseReportGenerator.getPolicyDisplayName(urls.get(i)));
79                  long count = 0;
80                  try {
81                      cmd = con.prepareStatement(
82                              "select count(*) from RawData where URI=? and "
83                              + "(UTCdatetime > ?) and (UTCdatetime < ?) ;");
84                      cmd.setString(1, urls.get(i));
85                      cmd.setLong(2, range.getStart().getTimeInMillis());
86                      cmd.setLong(3, range.getEnd().getTimeInMillis());
87                      rs = cmd.executeQuery();
88  
89                      try {
90                          if (rs.next()) {
91                              count = rs.getLong(1);
92                          }
93                      } catch (Exception ex) {
94                          log.log(Level.DEBUG, null, ex);
95                      }
96                  } catch (Exception ex) {
97                      log.log(Level.WARN, null, ex);
98                  } finally {
99                      DBUtils.safeClose(rs);
100                     DBUtils.safeClose(cmd);
101                 }
102 
103                 double d = range.getEnd().getTimeInMillis() - range.getStart().getTimeInMillis();
104                 double day = d / 86400000;
105                 double hours = d / 3600000;
106                 double min = d / 60000;
107                 double sec = d / 1000;
108 
109                 data.append("<tr><td>").append(url).append("</td><td>");
110                 data.append(count+"");
111                 data.append("</td><td>").append(format.format((double) ((double) count / day))).
112                         append("</td><td>").append(format.format((double) ((double) count / hours))).
113                         append("</td><td>").append(format.format((double) ((double) count / min))).
114                         append("</td><td>").append(format.format((double) ((double) count / sec))).
115                         append("</td></tr>");
116                 if (count > 0) {
117                     set.addValue((double) ((double) count / day), url, url);
118                 }
119 
120             }
121             chart = org.jfree.chart.ChartFactory.createBarChart(GetDisplayName(), "Service URL", "", set, PlotOrientation.HORIZONTAL, true, false, false);
122             data.append("</table>");
123             try {
124                 ChartUtilities.saveChartAsPNG(new File(path + getFilePathDelimitor() + "image_" + this.getClass().getSimpleName() + ".png"), chart, 1500, pixelHeightCalc(set.getRowCount()));
125             } catch (IOException ex) {
126                 log.log(Level.ERROR, "Error saving chart image for request", ex);
127             }
128 
129             data.append("<img src=\"image_").append(this.getClass().getSimpleName()).append(".png\">");
130             files.add(path + getFilePathDelimitor() + "image_" + this.getClass().getSimpleName() + ".png");
131         } catch (Exception ex) {
132             log.log(Level.ERROR, null, ex);
133         } finally {
134             DBUtils.safeClose(con);
135         }
136     }
