22  package org.miloss.fgsms.agents;
23  
24  import java.io.BufferedInputStream;
25  import java.io.ByteArrayInputStream;
26  import java.io.IOException;
27  import java.io.InputStream;
28  import java.io.PrintWriter;
29  import java.io.StringWriter;
30  import javax.servlet.ServletInputStream;
31  import javax.servlet.ServletOutputStream;
32  import javax.servlet.ServletRequest;
33  import javax.servlet.http.HttpServletRequest;
34  import javax.servlet.http.HttpServletRequestWrapper;
35  import javax.servlet.http.HttpServletResponse;
36  import javax.servlet.http.HttpServletResponseWrapper;
37  import org.miloss.fgsms.common.Constants;
38  
39  /**
40   *
41   * @author AO
42   */
43  public class InputStreamRequestWrapper extends HttpServletRequestWrapper {
44  
45      int BUFFER_SIZE = 1024;
46      private StringWriter sw = new StringWriter(BUFFER_SIZE);
47  
48      public InputStreamRequestWrapper(HttpServletRequest request) {
49          super(request);
50  
51          try {
52              BufferedInputStream buff = new BufferedInputStream(request.getInputStream());
53              if (buff.markSupported()) {
54                  buff.mark(1000000000);
55              }
56              byte[] buffer = new byte[1024];
57              int len = buff.read(buffer);
58              while (len > 0) {
59                  String data = new String(buffer, 0, len,Constants.CHARSET);
60                  sw.append(data);
61                  len = buff.read(buffer);
62              }
63              if (buff.markSupported()) {
64                  buff.reset();
65              }
66          } catch (Exception ex) {
67          }
68      }
69  
70      @Override
71      public ServletInputStream getInputStream() throws IOException {
72          ByteArrayInputStream in = new ByteArrayInputStream(sw.toString().getBytes(Constants.CHARSET));
73          return new ServletInputStreamWrapper(in);
74  
75      }
76  
77      public PrintWriter getWriter() throws IOException {
78          return new PrintWriter(sw);
79      }
80  
81      public ServletOutputStream getOutputStream() throws IOException {
82          throw new UnsupportedOperationException();
83      }
84  
85      /*
86       * returns the actual request message as a string
87       */
88      @Override
89      public String toString() {
90          return sw.toString();
91      }
92  }
