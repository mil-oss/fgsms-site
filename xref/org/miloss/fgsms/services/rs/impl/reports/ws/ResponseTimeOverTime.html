37  public class ResponseTimeOverTime extends BaseWebServiceReport {
38  
39      /**
40       * {@inheritDoc}
41       */
42      @Override
43      public String GetDisplayName() {
44          return "Response Time Over Time";
45      }
46  
47      /**
48       * {@inheritDoc}
49       */
50      @Override
51      public String GetHtmlFormattedHelp() {
52          return super.GetHtmlFormattedHelp() + "This represents the response time of a transactional web service, over time. For the chart X axis is time, Y axis is response time";
53      }
54  
55      @Override
56      public void generateReport(OutputStreamWriter data, List<String> urls, String path, List<String> files, TimeRange range, String currentuser, SecurityWrapper classification, WebServiceContext ctx) throws IOException {
57  
58          Connection con = Utility.getPerformanceDBConnection();
59          try {
60              PreparedStatement cmd = null;
61  
62              ResultSet rs = null;
63              JFreeChart chart = null;
64              data.append("<hr /><h2>").append(GetDisplayName()).append("</h2>");
65              data.append(GetHtmlFormattedHelp() + "<br />");
66              data.append("<table class=\"table table-hover\"><tr><th>URI</th><th>Average Response Time (ms)</th></tr>");
67  
68              TimeSeriesCollection col = new TimeSeriesCollection();
69              for (int i = 0; i < urls.size(); i++) {
70                  if (!isPolicyTypeOf(urls.get(i), PolicyType.TRANSACTIONAL)) {
71                      continue;
72                  }
73                  //https://github.com/mil-oss/fgsms/issues/112
74                  if (!UserIdentityUtil.hasReadAccess(currentuser, "getReport", urls.get(i), classification, ctx)) {
75                      continue;
76                  }
77  
78                  try {
79                      String url = Utility.encodeHTML(BaseReportGenerator.getPolicyDisplayName(urls.get(i)));
80                      data.append("<tr><td>").append(url).append("</td>");
81                      double average = 0;
82                      try {
83                          cmd = con.prepareStatement("select avg(responsetimems) from rawdata where uri=? and utcdatetime > ? and utcdatetime < ?;");
84                          cmd.setString(1, urls.get(i));
85                          cmd.setLong(2, range.getStart().getTimeInMillis());
86                          cmd.setLong(3, range.getEnd().getTimeInMillis());
87                          rs = cmd.executeQuery();
88  
89                          if (rs.next()) {
90                              average = rs.getDouble(1);
91  
92                          }
93                      } catch (Exception ex) {
94                          log.log(Level.WARN, null, ex);
95                      } finally {
96                          DBUtils.safeClose(rs);
97                          DBUtils.safeClose(cmd);
98                      }
99  
100                     data.append("<td>").append(format.format(average) + "").append("</td></tr>");
101 
102                     //ok now get the raw data....
103                     TimeSeriesContainer tsc = new TimeSeriesContainer();
104                     try {
105                         cmd = con.prepareStatement("select responsetimems,utcdatetime  from rawdata where uri=? and utcdatetime > ? and utcdatetime < ?;");
106                         cmd.setString(1, urls.get(i));
107                         cmd.setLong(2, range.getStart().getTimeInMillis());
108                         cmd.setLong(3, range.getEnd().getTimeInMillis());
109                         rs = cmd.executeQuery();
110                         while (rs.next()) {
111                             TimeSeries ts2 = tsc.Get(url, Millisecond.class);
112                             GregorianCalendar gcal = new GregorianCalendar();
113                             gcal.setTimeInMillis(rs.getLong(2));
114                             Millisecond m = new Millisecond(gcal.getTime());
115                             ts2.addOrUpdate(m, rs.getLong("responsetimems"));
116                         }
117                     } catch (Exception ex) {
118                         log.log(Level.WARN, null, ex);
119                     } finally {
120                         DBUtils.safeClose(rs);
121                         DBUtils.safeClose(cmd);
122                     }
123                     for (int ik = 0; ik < tsc.data.size(); ik++) {
124                         col.addSeries(tsc.data.get(ik));
125                     }
126 
127                 } catch (Exception ex) {
128                     log.log(Level.ERROR, "Error opening or querying the database.", ex);
129                 }
130 
131             }
132             chart = org.jfree.chart.ChartFactory.createTimeSeriesChart(GetDisplayName(), "Timestamp", "Time in ms", col, true, false, false);
133 
134             data.append("</table>");
135 
136             try {
137                 ChartUtilities.saveChartAsPNG(new File(path + getFilePathDelimitor() + "image_" + this.getClass().getSimpleName() + ".png"), chart, 1500, 400);
138                 data.append("<img src=\"image_").append(this.getClass().getSimpleName()).append(".png\">");
139                 files.add(path + getFilePathDelimitor() + "image_" + this.getClass().getSimpleName() + ".png");
140             } catch (IOException ex) {
141                 log.log(Level.ERROR, "Error saving chart image for request", ex);
142             }
143         } catch (Exception ex) {
144             log.log(Level.ERROR, null, ex);
145         } finally {
146             DBUtils.safeClose(con);
147         }
148     }
149 }
