20  package org.miloss.fgsms.auxsrv;
21  
22  import java.util.logging.Level;
23  import java.util.logging.Logger;
24  import org.miloss.fgsms.services.interfaces.common.PolicyType;
25  import org.miloss.fgsms.services.rs.impl.FgsmsReportGenerator;
26  import org.miloss.fgsms.sla.AuxHelper;
27  import org.miloss.fgsms.sla.SLACommon;
28  import org.quartz.*;
29  import org.quartz.JobExecutionContext;
30  import org.quartz.JobExecutionException;
31  
32  /**
33   **Quartz Job that kicks off the Report Generator
34   *
35   * @author AO
36   */
37  public class ReportGenScheduler implements StatefulJob {
38  
39      public void execute(JobExecutionContext jec) throws JobExecutionException {
40          try {
41              FgsmsReportGenerator gen = new FgsmsReportGenerator();
42  
43              gen.Fire(true, false);
44  
45              AuxHelper.TryUpdateStatus(true, "urn:fgsms:ReportGenerator:" + SLACommon.GetHostName(), "OK", true, PolicyType.STATUS, AuxHelper.UNSPECIFIED, SLACommon.GetHostName());
46          } catch (Exception ex) {
47              Logger.getLogger(ReportGenScheduler.class.getName()).log(Level.SEVERE, null, ex);
48              AuxHelper.TryUpdateStatus(false, "urn:fgsms:ReportGenerator:" + SLACommon.GetHostName(), ex.getMessage(), true, PolicyType.STATUS, AuxHelper.UNSPECIFIED, SLACommon.GetHostName());
49          }
50      }
51  
52  }
