32       */
33      void clearTransactionThreadId(long ThreadId);

35      /**

37       * specifically for mbean management
38       */
39      void forceNewDataPusherThread();

41      ConfigLoader getConfig();
42

44       * returns the lowercase value of the current hostname

46       * @return
47       */
48      String getHostName();

53       * gets the size of the current policy cache

55       * @return
56       */
57      int getPolicyCache();

60       * Attempts to find and return a service policy from the local cache. if it

62       * use this function as a memory footprint reducer.
63       *
64       * @param url
65       * @return service policy, possibly null
66       */
67      TransactionalWebServicePolicy getPolicyIfAvailable(String url);

69      /**
70       * @return the totalmessagesprocessed

72      long getTotalmessagesprocessed();

75       * Dependency Detection This should only be called in the context of a
76       * client handler seeing outbound transactions
77       */
78      String getTransactionThreadId(Long ThreadId);

80      /**
81       * turns all user identities as defined by the service policy, plus ip
82       * addresses
83       *
84       * @param p policy
85       * @param mc, current mesage correlation
86       * @return
87       */
88      ArrayList<String> getUserIdentities(TransactionalWebServicePolicy p, MessageCorrelator mc);

90      /**
91       * returns user identities via xpath
92       *
93       * @param xpaths
94       * @param message
95       * @return
96       */
97      ArrayList<String> getUsersfromXpath(ArrayOfXPathExpressionType xpaths, String message);
98  
99      /**
100      * gets the size of the current message map
101      *
102      * @return
103      */
104     int internalMessageMapSize();
105 
106     /**
107      * gets the size of the current queue
108      *
109      * @return
110      */
111     int outboundQueueSize();
112 
113     /**
114      * caches stored data and adds it to the message map. see note on
115      * agentclassname reference
116      *
117      *
118      * @param XMLrequest - may be null, the request content
119      * @param requestSize - size of the payload
120      * @param url - the requested URL that the client asked for
121      * http://localhost/path
122      * @param soapAction - SOAPAction or HTTP Method
123      * @param HttpUsername the username or DN of the requestor
124      * @param HashCode - very important, a unique identifier for this message.
125      * Use UUID.random
126      * @param headers - Transport layer headers of the request
127      * @param ipaddress - IP of the requestor
128      * @param agentclassname - the class of the agent that recorded the message.
129      * MUST CONTAIN "client" IF THIS IS A CLIENT or from a proxy TRANSACTION!!
130      * this is necessary for proper URL resolution
131      * @param relatedtransaction - if this transaction was also recorded
132      * elsewhere, pull this data from the http headers
133      * @param threadid - if this transaction was also recorded elsewhere, pull
134      * this data from the http headers. or generate a new one if it was not
135      * present using UUID.random
136      */
137     void processMessageInput(String XMLrequest, int requestSize, String url, String soapAction, String HttpUsername, String HashCode, HashMap headers, String ipaddress, String agentclassname, String relatedtransaction, String threadid);
138 
139 
140     /**
141      * wrapper to the main function This is normally only called when a CLIENT
142      * receives a response from a service
143      *
144      * @param HashCode
145      * @param responseXML
146      * @param responseSize
147      * @param isFault
148      * @param dod
149      * @param headers
150      * @param relatedTransaction
151      */
152     void processMessageOutput(String HashCode, String responseXML, int responseSize, boolean isFault, Long dod, HashMap headers, String relatedTransaction);
153 
154     /**
155      * pairs up a response with a request and adds it to the queue for
156      * transmission
157      *
158      * @param HashCode the UUID of the message that links this response to the
159      * request
160      * @param responseXML response payload
161      * @param responseSize
162      * @param isFault
163      * @param dod time of death, no longer used
164      * @param headers
165      */
166     void processMessageOutput(String HashCode, String responseXML, int responseSize, boolean isFault, Long dod, HashMap headers);
167 
168     /**
169      * this assumes that the request is already prepopulated with all required
170      * information according the policy for this item typically used for the
171      * offline persistent storage agent. As of RC6
172      *
173      * @param request
174      */
175     void processPreppedMessage(AddDataRequestMsg request);

179      */
180     void purgeMessageMap();

184      */

189      */
190     void purgePolicyCache();

193      * Any message in the Message Map older than the dead message interval will

195      * outbound queue
196      */
197     void removeDeadMessage();

221      */
222     void setTransactionThreadId(Long ThreadId, String id) throws Exception;

239      */
240     boolean shouldAgentRecordResponseContent(String requesturl);

242     void terminate();
243

245      * list of soap actions that should be ignored by agents
246      *
247      * @return
248      */
249     Set<String> getIgnoreList();

253      * @return the URLaddressMap
254      */
255     Map<String,String> getURLaddressMap();

262        void incMessagesProcessed(int size);
