89      public FgsmsSDK(Map<String, Object> context) throws Exception {
90          AuthMode mode = (AuthMode) context.get(AUTH_MODE);
91          if (mode == null) {
92              throw new Exception("Auth Mode is not defined");
93          }
94  
95          DataAccessService_Service svc = new DataAccessService_Service();
96          das = svc.getDASPort();
97          BindingProvider bp = (BindingProvider) das;
98          bp.getRequestContext().put(BindingProvider.ENDPOINT_ADDRESS_PROPERTY, context.get(DAS_ENDPOINT));
99  
100         if (mode == AuthMode.UsernamePassword) {
101             bp.getRequestContext().put(BindingProvider.USERNAME_PROPERTY, context.get(USERNAME_PROPERTY));
102             bp.getRequestContext().put(BindingProvider.PASSWORD_PROPERTY, context.get(PASSWORD_PROPERTY));
103         }
104 
105 
106 
107 
108         PolicyConfigurationService svc2 = new PolicyConfigurationService();
109         pcs = svc2.getPCSPort();
110         bp = (BindingProvider) pcs;
111         bp.getRequestContext().put(BindingProvider.ENDPOINT_ADDRESS_PROPERTY, context.get(PCS_ENDPOINT));
112 
113         if (mode == AuthMode.UsernamePassword) {
114             bp.getRequestContext().put(BindingProvider.USERNAME_PROPERTY, context.get(USERNAME_PROPERTY));
115             bp.getRequestContext().put(BindingProvider.PASSWORD_PROPERTY, context.get(PASSWORD_PROPERTY));
116         }
117 
118 
119 
120         ReportingService_Service svc3 = new ReportingService_Service();
121         rs = svc3.getReportingServicePort();
122         bp = (BindingProvider) rs;
123         bp.getRequestContext().put(BindingProvider.ENDPOINT_ADDRESS_PROPERTY, context.get(RS_ENDPOINT));
124 
125         if (mode == AuthMode.UsernamePassword) {
126             bp.getRequestContext().put(BindingProvider.USERNAME_PROPERTY, context.get(USERNAME_PROPERTY));
127             bp.getRequestContext().put(BindingProvider.PASSWORD_PROPERTY, context.get(PASSWORD_PROPERTY));
128         }
129 
130 
131 
132 
133         
134         AutomatedReportingService_Service svc4 = new AutomatedReportingService_Service();
135         ars = svc4.getAutomatedReportingServicePort();
136         bp = (BindingProvider) ars;
137         bp.getRequestContext().put(BindingProvider.ENDPOINT_ADDRESS_PROPERTY, context.get(ARS_ENDPOINT));
138 
139         if (mode == AuthMode.UsernamePassword) {
140             bp.getRequestContext().put(BindingProvider.USERNAME_PROPERTY, context.get(USERNAME_PROPERTY));
141             bp.getRequestContext().put(BindingProvider.PASSWORD_PROPERTY, context.get(PASSWORD_PROPERTY));
142         }
143 
144 
145 
146 
147 
148 
149 
150         StatusServiceService svc5 = new StatusServiceService();
151         ss = svc5.getStatusServicePort();
152         bp = (BindingProvider) ss;
153         bp.getRequestContext().put(BindingProvider.ENDPOINT_ADDRESS_PROPERTY, context.get(SS_ENDPOINT));
154 
155         if (mode == AuthMode.UsernamePassword) {
156             bp.getRequestContext().put(BindingProvider.USERNAME_PROPERTY, context.get(USERNAME_PROPERTY));
157             bp.getRequestContext().put(BindingProvider.PASSWORD_PROPERTY, context.get(PASSWORD_PROPERTY));
158         }
159 
160 
161 
162         if (context.get(CLASSIFICATION_LEVEL) != null) {
163             sc.setClassification((ClassificationType) context.get(CLASSIFICATION_LEVEL));
164         }
165         if (context.get(CLASSIFICATION_CAVEAT) != null) {
166             sc.setCaveats((String) context.get(CLASSIFICATION_CAVEAT));
167         }
168 
169     }

172         GetMonitoredServiceListRequestMsg req = new GetMonitoredServiceListRequestMsg();

175         if (monitoredServiceList != null && monitoredServiceList.getServiceList() != null && monitoredServiceList.getServiceList() != null) {
176             return monitoredServiceList.getServiceList().getServiceType();
177         }
178         return null;
179     }

302     SetServicePolicyResponseMsg StartProcessMonitor(String process, String machine, String domain, String displayname, String machineurl, long datattl, List<SLA> slas) throws Exception {
303         SetServicePolicyRequestMsg preq = new SetServicePolicyRequestMsg();
304         preq.setClassification(sc);
305         preq.setURL("urn:" + machine.toLowerCase() + ":" + process);
306         ProcessPolicy pp = new ProcessPolicy();
307         pp.setMachineName(machine.toLowerCase());
308         pp.setDomainName(domain);
309         pp.setDisplayName(displayname);
310         pp.setPolicyType(PolicyType.PROCESS);
311         pp.setAgentsEnabled(true);
312         pp.setURL("urn:" + machine.toLowerCase() + ":" + process);
313         pp.setParentObject(machineurl);
314         pp.setDataTTL(DatatypeFactory.newInstance().newDuration(datattl));
315         //30L * 24L * 60L * 60L * 1000L));
316         Utility.addStatusChangeSLA(pp);
317         if (slas != null) {
318             pp.getServiceLevelAggrements().getSLA().addAll(slas);
319         }
320         preq.setPolicy(pp);
321         return pcs.setServicePolicy(preq);
322     }

357     public void GetCSVExportNIC(List<String> urls, String NICID, long start, long end, String filename) throws Exception {
358         ExportCSVDataRequestMsg req = new ExportCSVDataRequestMsg();
359         req.setClassification(sc);
360         req.setAllServices(false);
361         req.setExportType(ExportRecordsEnum.MACHINE);
362         req.setRange(new TimeRange());
363         req.getRange().setStart(CreateXmlGreg(start));
364         req.getRange().setEnd(CreateXmlGreg(end));
365         req.getURLs().addAll(urls);
366         ExportDataToCSVResponseMsg exportDataToCSV = this.rs.exportDataToCSV(req);
367         WriteBytes(exportDataToCSV.getZipFile(), filename);
368     }

371     private Calendar CreateXmlGreg(long time) throws DatatypeConfigurationException {
372         if (df == null) {
373             df = DatatypeFactory.newInstance();
374         }
375         GregorianCalendar gcal = new GregorianCalendar();
376         gcal.setTimeInMillis(time);
377         return (gcal);
378     }

380     private void WriteBytes(byte[] zipFile, String filename) throws Exception {
381         FileOutputStream fos = new FileOutputStream(filename);
382         fos.write(zipFile);
383         fos.flush();
384         fos.close();
385     }
