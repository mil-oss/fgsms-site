22  package org.miloss.fgsms.agents;
23  
24  import org.miloss.fgsms.common.Constants;
25  import java.util.Collections;
26  import java.util.Set;
27  import javax.xml.namespace.QName;
28  import javax.xml.soap.SOAPException;
29  import javax.xml.ws.handler.MessageContext;
30  import javax.xml.ws.handler.soap.SOAPHandler;
31  import javax.xml.ws.handler.soap.SOAPMessageContext;
32  import org.apache.log4j.Level;
33  import org.miloss.fgsms.common.Logger;;
34  
35  /**
36   *
37   * @author AO
38   */
39  public class JbossWSClientAgent implements SOAPHandler<SOAPMessageContext> {
40  
41      private Logger log;
42  
43      public JbossWSClientAgent() {
44          log = Logger.getLogger(org.miloss.fgsms.common.Constants.LoggerName);
45      }
46  
47      public boolean handleMessage(SOAPMessageContext messageContext) {
48          long start = System.currentTimeMillis();
49          // MessageProcessor obj = MessageProcessor.getSingletonObject();
50          Boolean isOutbound = (Boolean) messageContext.get(SOAPMessageContext.MESSAGE_OUTBOUND_PROPERTY);
51  
52          if (isOutbound) {
53  
54              try {
55                  ProcessOutboundMessage(messageContext);
56              } catch (SOAPException ex) {
57                  log.log(Level.WARN, "fgsms Error processing inbound message", ex);
58                  return true;
59              }
60          } else {
61              try {
62                  ProcessInboundMessage(messageContext, false);
63              } catch (SOAPException ex) {
64                  log.log(Level.WARN, "fgsms Error processing outbound message", ex);
65              }
66  
67          }
68  
69  
70          // log.log(Level.WARNING, "fgsms Timer, handleMessage " + (System.currentTimeMillis() - start));
71          return true;
72      }
73  
74      public Set<QName> getHeaders() {
75          return Collections.EMPTY_SET;
76      }
77  //this will not be called if there is a problem with the proxy, we'll need to add something to purge old records
78      
79      public boolean handleFault(SOAPMessageContext messageContext) {
80          Boolean isOutbound = (Boolean) messageContext.get(SOAPMessageContext.MESSAGE_OUTBOUND_PROPERTY);
81  
82          if (isOutbound) {
83  
84              try {
85                  ProcessOutboundMessage(messageContext);
86              } catch (SOAPException ex) {
87                  log.log(Level.WARN, "fgsms Error processing inbound message", ex);
88              }
89          } else {
90              try {
91                  ProcessInboundMessage(messageContext, true);
92              } catch (SOAPException ex) {
93                  log.log(Level.WARN, "fgsms Error processing outbound message", ex);
94              }
95  
96          }
97          return true;
98      }
99  
100     public void close(MessageContext context) {
101     }
102 
103     private void ProcessOutboundMessage(SOAPMessageContext messageContext) throws SOAPException {
104 
105         //this is for an outbound request
106 
107         JbossWSCommonMessageHandler.ProcessRequest(messageContext, true, this.getClass().getCanonicalName());
108 
109     }
110 
111 
112 
113     private void ProcessInboundMessage(SOAPMessageContext messageContext, boolean fault) throws SOAPException {
114 
115         /*
116          * This is an inbound reply message!
117          *
118          */
119         JbossWSCommonMessageHandler.ProcessResponse(messageContext, fault, true, this.getClass().getCanonicalName());
120 /*
121  * This is an inbound reply message!
122  *
123  */
124 
125     }
126 }
