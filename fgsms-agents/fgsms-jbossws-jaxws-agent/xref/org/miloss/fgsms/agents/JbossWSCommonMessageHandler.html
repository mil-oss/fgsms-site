434     public static void ProcessResponse(SOAPMessageContext messageContext, boolean fault, boolean client, String classname) {
435         long now = System.currentTimeMillis();
436         log.log(Level.TRACE, "JbossWSCommonMessageHandler.ProcessResponse0 timer: " + (System.currentTimeMillis() - now) + " thread:" + Thread.currentThread().getId());
437 
438         boolean isoneway = messageContext.containsKey(org.miloss.fgsms.common.Constants.oneway);
439         //if client, send in data including transport
440         //if service, send in data for only processing time
441         log.log(Level.DEBUG, "Current VM Memory : total = " + Runtime.getRuntime().totalMemory() + " free = " + Runtime.getRuntime().freeMemory());
442         
443         UUID id = (UUID) messageContext.get(org.miloss.fgsms.common.Constants.key);
444         String requrl = (String) messageContext.get(org.miloss.fgsms.common.Constants.urlKEY);
445         boolean ok = true;
446         if (id == null) {
447             log.log(Level.DEBUG, "fgsms reply message did not have context variable " + org.miloss.fgsms.common.Constants.key + " added. This transaction will be ignored.");
448             ok = false;
449         }
450         if (Utility.stringIsNullOrEmpty(requrl)) {
451             log.log(Level.DEBUG, "fgsms reply message did not have context variable " + org.miloss.fgsms.common.Constants.urlKEY + " added. This transaction will be ignored.");
452             ok = false;
453         }
454         String relatedTransaction = null;
455         if (!ok) {
456             if (id != null) {
457                 MessageProcessor.getSingletonObject().removeFromQueue(id);
458             }
459             return;
460         }
461         log.log(Level.TRACE, "JbossWSCommonMessageHandler.ProcessResponse1 timer: " + (System.currentTimeMillis() - now) + " thread:" + Thread.currentThread().getId());
462         HashMap headervalues = new HashMap();
463 
464 
465         if (client && !isoneway) {   //response
466             try {
467                 Map headers = (Map) messageContext.get(messageContext.HTTP_RESPONSE_HEADERS);
468                 if (headers != null) {
469                     log.log(Level.DEBUG, "client received " + headers.size() + " transaction " + id.toString());
470                     //  Iterator it = headers.keySet().iterator();
471                     Iterator<Entry<String, List<String>>> it = headers.entrySet().iterator();
472                     while (it.hasNext()) {
473                         Entry<String, List<String>> s = (Entry<String, List<String>>) it.next();
474                         //  log.log(Level.DEBUG, " client header " + s + " = " + headers.get(s).toString());
475                         headervalues.put(s.getKey(), s.getValue());
476                     }
477                     if (headers.containsKey(org.miloss.fgsms.common.Constants.relatedtransactionKey)) {
478                         relatedTransaction = (String) headers.get(org.miloss.fgsms.common.Constants.relatedtransactionKey);
479                     }
480                 }
481                 /*
482                  * if
483                  * (messageContext.getMessage().getSOAPHeader().getChildElements(new
484                  * QName("org.miloss.fgsms.agent",
485                  * "fgsmsAgent")).hasNext()) { log.log(Level.INFO, "fgsms
486                  * message has been flagged as being recorded by another agent
487                  * via soap header. It will be ignored");
488                  * MessageProcessor.RemoveFromQueue(id); return; }
489                  */
490             } catch (Exception ex) {
491                 log.log(Level.ERROR, "Unexpected error caught when searching for fgsms soap header", ex);
492             }
493             if (Utility.stringIsNullOrEmpty(relatedTransaction)) {
494                 relatedTransaction = DependencyHelper.getRelatedMessageIdFromSoapHeader(messageContext);
495             }
496         }
497         log.log(Level.TRACE, "JbossWSCommonMessageHandler.ProcessResponse2 timer: " + (System.currentTimeMillis() - now) + " thread:" + Thread.currentThread().getId());
498         if (!client && !isoneway) {
499             //service response, insert fgsms headers
500             ArrayList<String> l = new ArrayList<String>();
501             l.add(id.toString());
502             ArrayList<String> threadlist = new ArrayList<String>();
503             threadlist.add(MessageProcessor.getSingletonObject().getTransactionThreadId(Thread.currentThread().getId()));
504             java.util.Map<java.lang.String, java.util.List<java.lang.String>> headers = (Map<String, List<String>>) messageContext.get(messageContext.HTTP_RESPONSE_HEADERS);
505             if (MessageProcessor.getSingletonObject().isDependencyInjectionEnabled()) {
506                 DependencyHelper.insertRelatedMessageHeader(id.toString(), messageContext);
507                 DependencyHelper.insertThreadIdHeader(threadlist.get(0), messageContext);
508                 if (headers != null) {
509                     headers.put(org.miloss.fgsms.common.Constants.relatedtransactionKey, l);
510                 } else {
511                     headers = new HashMap<String, List<String>>();
512                     headers.put(org.miloss.fgsms.common.Constants.relatedtransactionKey, l);
513                     headers.put(org.miloss.fgsms.common.Constants.transactionthreadKey, threadlist);
514                     messageContext.put(messageContext.HTTP_RESPONSE_HEADERS, headers);
515                     //log.log(Level.WARN, "fgsms response header map is null");
516                 }
517             }
518             //Iterator it = headers.keySet().iterator();
519             Iterator<Entry<String, List<String>>> it = headers.entrySet().iterator();
520             while (it.hasNext()) {
