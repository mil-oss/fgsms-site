122     private void SetupPCSEndpoint() {
123       
124         PolicyConfigurationService c = new PolicyConfigurationService();
125         pcsPort = c.getPCSPort();
126         BindingProvider bp = (BindingProvider) pcsPort;
127         bp.getRequestContext().put(BindingProvider.ENDPOINT_ADDRESS_PROPERTY, (String) prop.get("PCS"));
128         bp.getRequestContext().put(BindingProvider.USERNAME_PROPERTY, (String) prop.get("fgsmsadminuser"));
129         bp.getRequestContext().put(BindingProvider.PASSWORD_PROPERTY, (String) prop.get("fgsmsadminpass"));
130     }

132     private void SetServicePolicy(String firsthop_url, PolicyType pt) throws Exception {
133 
134 
135         SetServicePolicyRequestMsg req = new SetServicePolicyRequestMsg();
136         req.setClassification(new SecurityWrapper());
137         req.setURL(firsthop_url);
138 
139         ServicePolicy sp = null;
140         switch (pt) {
141             case TRANSACTIONAL:
142                 TransactionalWebServicePolicy tp = new TransactionalWebServicePolicy();
143                 tp.setRecordFaultsOnly(false);
144                 tp.setRecordRequestMessage(false);
145                 tp.setRecordResponseMessage(false);
146                 tp.setBuellerEnabled(false);
147                 tp.setHealthStatusEnabled(false);
148                 tp.setURL(firsthop_url);
149                 tp.setDomainName("unspecified");
150                 tp.setMachineName(Utility.getHostName());
151                 sp = tp;
152                 break;
153             case MACHINE:
154                 MachinePolicy mp = new MachinePolicy();
155                 sp = mp;
156                 break;
157             case PROCESS:
158                 ProcessPolicy pp = new ProcessPolicy();
159                 sp = pp;
160                 break;
161             case STATISTICAL:
162                 StatisticalServicePolicy bps = new StatisticalServicePolicy();
163                 sp = bps;
164                 break;
165             case STATUS:
166                 StatusServicePolicy ssp = new StatusServicePolicy();
167                 sp = ssp;
168         }
169         sp.setPolicyType(pt);
170         sp.setServiceLevelAggrements(addSLAPlugin());
171         req.setPolicy(sp);
172         req.setURL(firsthop_url);
173         pcsPort.setServicePolicy(req);
174     }

176     private ArrayOfSLA addSLAPlugin() {
177         
178         ArrayOfSLA slas = new ArrayOfSLA();
179         SLA sla = new SLA();
180         sla.setAction(new ArrayOfSLAActionBaseType());
181         SLAAction sa = new SLAAction();
182         sa.setImplementingClassName(CustomSLAAction.class.getCanonicalName());
183 
184         sla.getAction().getSLAAction().add(sa);
185         sla.setGuid(UUID.randomUUID().toString());
186         SLARuleGeneric rule = new SLARuleGeneric();
187         rule.setClassName(CustomSLARule.class.getCanonicalName());
188         rule.setProcessAt(RunAtLocation.FGSMS_SERVER);
189 
190         sla.setRule(rule);
191 
192         slas.getSLA().add(sla);
193         return (slas);
194     }

196     private void SendData(String url) throws Exception {
197         List<AddDataRequestMsg> req = new ArrayList<AddDataRequestMsg>();
198         //for (int i = 0; i < expectedcount; i++) {
199         AddDataRequestMsg r = new AddDataRequestMsg();
200         r.setSuccess(false);
201         r.setAgentType("jUnit");
202         r.setAction("test");
203         r.setClassification(new SecurityWrapper());
204         GregorianCalendar gcal = new GregorianCalendar();
205         gcal.setTimeInMillis(System.currentTimeMillis());
206         r.setRecordedat(gcal);
207         r.setRequestSize(100);
208         r.setRequestURI(url);
209         r.setResponseSize(100);
210         r.setResponseTime(100);
211         r.setServiceHost(Utility.getHostName());
212         r.setTransactionID(UUID.randomUUID().toString());
213         r.setURI(url);
214         req.add(r);
215         //}
216         DCS dcs = cfg.getDcsport();
217         BindingProvider bp = (BindingProvider) dcs;
218         bp.getRequestContext().put(BindingProvider.ENDPOINT_ADDRESS_PROPERTY, cfg.getDCS_URLS().get(0));
219         bp.getRequestContext().put(BindingProvider.USERNAME_PROPERTY, cfg.getUsername());
220         bp.getRequestContext().put(BindingProvider.PASSWORD_PROPERTY, Utility.DE(cfg.getPassword()));
221         // System.out.println("Delivering " + expectedcount + " msgs to dcs");
222         dcs.addMoreData(req);
223     }

225     private void ConfigureCallbackPlugin(String url) throws Exception {
226         SetGeneralSettingsRequestMsg req = new SetGeneralSettingsRequestMsg();
227         req.setClassification(new SecurityWrapper());
228         KeyNameValueEnc keyNameValueEnc = new KeyNameValueEnc();
229         keyNameValueEnc.setShouldEncrypt(false);
230         KeyNameValue knv = new KeyNameValue();
231         knv.setPropertyKey("SDK.Samples.AlertRecieverService");
232         knv.setPropertyName("URL");
233         knv.setPropertyValue(url);
234         keyNameValueEnc.setKeyNameValue(knv);
235         req.getKeyNameValueEnc().add(keyNameValueEnc);
236         pcsPort.setGeneralSettings(req);
237     }
238 
239     private void UnconfigureCallbackPlugin() throws Exception {
240         RemoveGeneralSettingsRequestMsg req = new RemoveGeneralSettingsRequestMsg();
241 
242         KeyNameValue knv = new KeyNameValue();
243         knv.setPropertyKey("SDK.Samples.AlertRecieverService");
244         knv.setPropertyName("URL");
245         req.setClassification(new SecurityWrapper());
246         req.getKeyNameValue().add(knv);
247         pcsPort.removeGeneralSettings(req);
248 
249 
250     }
