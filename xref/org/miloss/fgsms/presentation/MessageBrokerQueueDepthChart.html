24  import de.laures.cewolf.DatasetProduceException;
25  import de.laures.cewolf.DatasetProducer;
26  import java.io.Serializable;
27  import java.util.Date;
28  import java.util.Map;
29  
30  import org.miloss.fgsms.services.interfaces.dataaccessservice.*;
31  import org.apache.log4j.Level;
32  import org.miloss.fgsms.common.Logger;;
33  import org.jfree.data.category.DefaultCategoryDataset;
34  
35  /**
36   *
37   * @author AO
38   */
39  public class MessageBrokerQueueDepthChart implements DatasetProducer, Serializable {
40  
41      public Object produceDataset(Map params) throws DatasetProduceException {
42          DefaultCategoryDataset data = new DefaultCategoryDataset();
43          try{
44          GetCurrentBrokerDetailsResponseMsg res = (GetCurrentBrokerDetailsResponseMsg) params.get("fgsms.data");
45  
46          for (int i = 0; i < res.getQueueORtopicDetails().size(); i++) {
47              if (res.getQueueORtopicDetails().get(i).getItemtype().equalsIgnoreCase("queue")) {
48                  //data.addValue(set.stuff.get(i).success, set.stuff.get(i).action + " Success", set.stuff.get(i).action);
49                  long k = res.getQueueORtopicDetails().get(i).getQueueDepth();
50                 // k = (long) (Math.random() * 100);
51                  data.addValue( k,res.getQueueORtopicDetails().get(i).getCanonicalname(),res.getQueueORtopicDetails().get(i).getCanonicalname());
52              }
53          }
54          }
55          catch (Exception ex)
56          {
57              LogHelper.getLog().log(Level.WARN, "error building chart for message broker queue depth", ex);
58          }
59          return data;
60      }
61  
62      public boolean hasExpired(Map params, Date since) {
63          return (System.currentTimeMillis() - since.getTime()) > 5000;
64      }
65  
66      public String getProducerId() {
67         return "fgsms.MessageBrokerQueueDepthChart";
68      }
69  }
