285     public static void CreatePolicy(String uri, FederationPolicyCollection fedpolicies) {
286         Connection con = Utility.getConfigurationDBConnection();
287         Connection perf = Utility.getPerformanceDBConnection();
288         try {
289 
290             PreparedStatement com = con.prepareStatement("delete from servicepolicies where uri=?;");
291             com.setString(1, uri);
292             com.execute();
293             com = con.prepareStatement(
294                     "INSERT INTO servicepolicies("
295                     + "            uri, datattl, " //5
296                     + "            buellerenabled, " //3
297                     + "            policytype, xmlpolicy,hasfederation)" //1
298                     + "    VALUES (?, ?, ?, ?, ?,?);");
299             com.setString(1, uri);
300             com.setLong(2, 500000);
301             com.setBoolean(3, true);
302             com.setInt(4, PolicyType.TRANSACTIONAL.ordinal());
303             TransactionalWebServicePolicy twp = new TransactionalWebServicePolicy();
304             twp.setURL(uri);
305             twp.setFederationPolicyCollection(fedpolicies);
306             twp.setDescription("jUnit Test Description");
307             DatatypeFactory fac = DatatypeFactory.newInstance();
308             twp.setDataTTL(fac.newDuration(10000000));
309             Marshaller m = Utility.getSerializationContext().createMarshaller();
310             StringWriter sw = new StringWriter();
311             m.marshal(twp, sw);
312 
313             String s = sw.toString();
314             com.setBytes(5, s.getBytes());
315 
316             com.setBoolean(6, fedpolicies != null && !fedpolicies.getFederationPolicy().isEmpty());
317 
318             com.execute();
319             com.close();
320             //System.out.println("attempting config connection close");
321             con.close();
322             //System.out.println("config connection closed");
323 
324             com = perf.prepareStatement("INSERT INTO rawdatatally(uri, success, faults, slafault)    VALUES (?, ?, ?, ?);");

326             com.setLong(2, 0);
327             com.setLong(3, 0);
328             com.setLong(4, 0);
329             com.execute();
330             com.close();
331             //System.out.println("attempting perf connection close");
332             perf.close();
333             //System.out.println("perf connection closed");
334         } catch (Exception ex) {
