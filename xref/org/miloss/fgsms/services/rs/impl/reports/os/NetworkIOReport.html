39  public class NetworkIOReport extends BaseReportGenerator {
40  
41      @Override
42      public String GetDisplayName() {
43          return "Network I/O rates by Machine";
44      }
45  
46      @Override
47      public String GetHtmlFormattedHelp() {
48          return "This represents the network throughput rates of a machine over time as reported by the operating system, hypervisor, and/or network card. For the chart, X axis is time, Y is I/O rate";
49      }
50  
51      @Override
52      public List<PolicyType> GetAppliesTo() {
53          ArrayList<PolicyType> ret = new ArrayList<PolicyType>();
54          ret.add(PolicyType.MACHINE);
55          return ret;
56      }
57  
58      @Override
59      public void generateReport(OutputStreamWriter data, List<String> urls, String path, List<String> files, TimeRange range, String currentuser, SecurityWrapper classification, WebServiceContext ctx) throws IOException {
60  
61          Connection con = Utility.getPerformanceDBConnection();
62          try{
63          PreparedStatement cmd = null;
64          ResultSet rs = null;
65          DefaultCategoryDataset set = new DefaultCategoryDataset();
66          JFreeChart chart = null;
67          data.append("<hr /><h2>").append(GetDisplayName()).append("</h2>");
68          data.append("This represents the network throughput rates of a machine over time.<br />");
69          data.append("<table class=\"table table-hover\"><tr><th>URI</th><th>Average Send Rate</th><th>Average Recieve Rate</th></tr>");
70  
71          TimeSeriesCollection col = new TimeSeriesCollection();
72          for (int i = 0; i < urls.size(); i++) {
73              if (!isPolicyTypeOf(urls.get(i), PolicyType.MACHINE)) {
74                  continue;
75              }
76              //https://github.com/mil-oss/fgsms/issues/112
77              if (!UserIdentityUtil.hasReadAccess(currentuser, "getReport", urls.get(i), classification, ctx)) {
78                  continue;
79              }
80  String url = Utility.encodeHTML(BaseReportGenerator.getPolicyDisplayName(urls.get(i)));
81              data.append("<tr><td>").append(url).append("</td>");
82              double average = 0;
83              try {
84                  cmd = con.prepareStatement("select avg(sendkbs) from rawdatanic where uri=? and utcdatetime > ? and utcdatetime < ?;");
85                  cmd.setString(1, urls.get(i));
86                  cmd.setLong(2, range.getStart().getTimeInMillis());
87                  cmd.setLong(3, range.getEnd().getTimeInMillis());
88                  rs = cmd.executeQuery();
89  
90                  if (rs.next()) {
91                      average = rs.getDouble(1);
92  
93                  }
94              } catch (Exception ex) {
95                  log.log(Level.WARN, null, ex);
96              } finally {
97                  DBUtils.safeClose(rs);
98                  DBUtils.safeClose(cmd);
99              }
100 
101             data.append("<td>").append(average + "").append("</td>");
102             average = 0;
103             try {
104                 cmd = con.prepareStatement("select avg(receivekbs) from rawdatanic where uri=? and utcdatetime > ? and utcdatetime < ?;");
105                 cmd.setString(1, urls.get(i));
106                 cmd.setLong(2, range.getStart().getTimeInMillis());
107                 cmd.setLong(3, range.getEnd().getTimeInMillis());
108                 rs = cmd.executeQuery();
109 
110                 if (rs.next()) {
111                     average = rs.getDouble(1);
112 
113                 }
114             } catch (Exception ex) {
115                 log.log(Level.WARN, null, ex);
116             } finally {
117                 DBUtils.safeClose(rs);
118                 DBUtils.safeClose(cmd);
119             }
120             data.append("<td>").append(average + "").append("</td></tr>");
121 
122             //ok now get the raw data....
123             TimeSeriesContainer tsc = new TimeSeriesContainer();
124             try {
125                 cmd = con.prepareStatement("select receivekbs, sendkbs, utcdatetime, nicid from rawdatanic where uri=? and utcdatetime > ? and utcdatetime < ?;");
126                 cmd.setString(1, urls.get(i));
127                 cmd.setLong(2, range.getStart().getTimeInMillis());
128                 cmd.setLong(3, range.getEnd().getTimeInMillis());
129                 rs = cmd.executeQuery();
130 
131                 while (rs.next()) {
132 
133                     TimeSeries ts = tsc.Get(url + " " + rs.getString("nicid") + " RX", Millisecond.class);
134                     TimeSeries ts2 = tsc.Get(url + " " + rs.getString("nicid") + " TX", Millisecond.class);
135                     GregorianCalendar gcal = new GregorianCalendar();
136                     gcal.setTimeInMillis(rs.getLong(3));
137                     Millisecond m = new Millisecond(gcal.getTime());
138                     ts.addOrUpdate(m, rs.getLong(1));
139                     ts2.addOrUpdate(m, rs.getLong(2));
140                 }
141             } catch (Exception ex) {
142                 log.log(Level.WARN, null, ex);
143             } finally {
144                 DBUtils.safeClose(rs);
145                 DBUtils.safeClose(cmd);
146             }
147             for (int ik = 0; ik < tsc.data.size(); ik++) {
148                 col.addSeries(tsc.data.get(ik));
149             }
150 
151         }
152 
153         chart = org.jfree.chart.ChartFactory.createTimeSeriesChart(GetDisplayName(), "Timestamp", "Rate", col, true, false, false);
154 
155         data.append("</table>");
156         try {
157             // if (set.getRowCount() != 0) {
158             ChartUtilities.saveChartAsPNG(new File(path + getFilePathDelimitor() + "image_" + this.getClass().getSimpleName() + ".png"), chart, 1500, 400);
159             data.append("<img src=\"image_").append(this.getClass().getSimpleName()).append(".png\">");
160             files.add(path + getFilePathDelimitor() + "image_" + this.getClass().getSimpleName() + ".png");
161             // }
162         } catch (IOException ex) {
163             log.log(Level.ERROR, "Error saving chart image for request", ex);
164         }
165          } catch (Exception ex) {
166             log.log(Level.ERROR, null, ex);
167         } finally {
168             DBUtils.safeClose(con);
169         }
170 
171     }
