35       * Called as the message flows towards the transport.
36       * this is really on output!!!
37       * http://docs.jboss.org/jbossesb/docs/4.2MR3/javadoc/esb/index.html?org/jboss/soa/esb/filter/InputOutputFilter.html

43      public org.jboss.soa.esb.message.Message onInput(org.jboss.soa.esb.message.Message msg,
44              java.util.Map<java.lang.String, java.lang.Object> params)
45              throws org.jboss.soa.esb.couriers.CourierException {
46          log.log(Level.INFO, "Current VM Memory : total = " + Runtime.getRuntime().totalMemory() + " free = " + Runtime.getRuntime().freeMemory());
47  
48  
49          for (int i = 0; i < msg.getProperties().getNames().length; i++) {
50              log.log(Level.DEBUG, "key = " + msg.getProperties().getNames()[i] + " value = " + msg.getProperties().getProperty(msg.getProperties().getNames()[i]));
51          }
52          if (params != null) {
53              Iterator<String> iterator = params.keySet().iterator();
54              while (iterator.hasNext()) {
55                  String s = iterator.next();
56                  log.log(Level.DEBUG, "pkey = " + s + " pvalue = " + params.get(s).toString());
57              }
58          }
59  
60          //an output message could be a request, service or web app as a client OR it could be a response from this service, handle this
61  
62  
63          //Serializable obj = (Serializable) msg.getBody().get();
64          String id = "";
65          try {
66              id = (String) msg.getProperties().getProperty("fgsms.TransactionID");
67          } catch (Exception ex) {
68              log.log(Level.ERROR, "fgsms Agent for JbossESB, could not obtain the transaction id, this was unexpected." + ex.getLocalizedMessage());
69              return msg;
70          }
71          if (Utility.stringIsNullOrEmpty(id)) {
72              log.log(Level.ERROR, "fgsms Agent for JbossESB, outbound message does not have transaction id, this was unexpected.");
73              return msg;
74          }
75  
76          //grab the original url from the gateway, useful but not fullproof
77          String url = (String) msg.getProperties().getProperty("org.jboss.soa.esb.gateway.original.url");
78          if (Utility.stringIsNullOrEmpty(url)) {
79              try {
80                  //alternate method, but should always work
81                  url = msg.getHeader().getCall().getTo().getURI().toString();
82                  if (url.startsWith("invm://")) {
83                      log.log(Level.WARN, "fgsms Agent for JbossESB, outbound message is internal esb traffic, ignoring " + url);
84                      return msg;
85                  }
86                  log.log(Level.WARN, "fgsms Agent for JbossESB, outbound message for " + url + " via SOAP To field.");
87              } catch (Exception ex) {
88                  log.log(Level.ERROR, "fgsms Agent for JbossESB, unable to determine the URL or TO field from the message, this message will be ignored");
89                  return msg;
90              }
91          } else {
92              log.log(Level.INFO, "fgsms Agent for JbossESB, outbound message for " + url + " via transport.");
93          }
94          // search for endswith _reply to identify reply messages from queues
95          if (url.endsWith("_reply")) {
96          }
97  
98  
99          Integer msgsize = null;
100         try {
101             msgsize = (Integer) msg.getProperties().getProperty("org.jboss.soa.esb.message.byte.size");
102             log.log(Level.INFO, "fgsms Agent for JbossESB, outbound message size " + msgsize + " via long.");
103         } catch (Exception ex) {
104         }
105         try {
106             msgsize = Integer.parseInt((String) msg.getProperties().getProperty("org.jboss.soa.esb.message.byte.size"));
107             log.log(Level.INFO, "fgsms Agent for JbossESB, outbound message size " + msgsize + " via string.");
108         } catch (Exception ex) {
109         }
110 
111 
112         boolean fault = false;
113         //getFault 'should' be for comms error, not app error
114         if (msg.getFault() != null) {
115             if (!Utility.stringIsNullOrEmpty(msg.getFault().getReason())
116                     || msg.getFault().getCode() != null
117                     || msg.getFault().getCause() != null) //faulted msg
118             {
119                 fault = true;
120                 log.log(Level.WARN, "fgsms, this message to " + url + " transaction id:" + id.toString() + " has faulted.");
121             }
122         }
123 
124 //        Long dod = (Long) params.get("org.jboss.soa.esb.message.time.dod");
125         //      if (dod == null) {
126         Long dod = System.currentTimeMillis();
127         //    }
128 
129         if (msg.getAttachment().getNamedCount() > 0 || msg.getAttachment().getUnnamedCount() > 0) {
130             log.log(Level.INFO, "fgsms Agent for JbossESB, outbound message has attachments, named: " + msg.getAttachment().getNamedCount() + " unnamed:" + msg.getAttachment().getUnnamedCount());
131             int count = (msg.getAttachment().getNamedCount() + msg.getAttachment().getUnnamedCount());
132             for (int i = 0; i < count; i++) {
133                 log.log(Level.INFO, "fgsms Agent for JbossESB, outbound message has attachments " + i + " type: " + msg.getAttachment().itemAt(i).getClass().getCanonicalName());
134             }
135         }
136 
137 
138         String body = "";
139         try {
140             if (msg.getBody() instanceof TextBody) {
141                 TextBody t = (TextBody) msg.getBody();
142                 if (t != null) {
143                     body = t.getText();
144                     body += t.get().toString();
145                 }
146             }
147             if (msg.getBody() instanceof org.jboss.internal.soa.esb.message.format.xml.BodyImpl) {
148                 org.jboss.internal.soa.esb.message.format.xml.BodyImpl t = (org.jboss.internal.soa.esb.message.format.xml.BodyImpl) msg.getBody();
149                 log.log(Level.DEBUG, "body get is of type " + t.get().getClass().getCanonicalName());
150                 body = t.toString();//new String(t.getContents());
151                 body += t.get().toString();
152             }
153             if (msg.getBody() instanceof org.jboss.internal.soa.esb.message.format.serialized.BodyImpl) {
154                 org.jboss.internal.soa.esb.message.format.serialized.BodyImpl t = (org.jboss.internal.soa.esb.message.format.serialized.BodyImpl) msg.getBody();
155                 log.log(Level.DEBUG, "body get is of type " + t.get().getClass().getCanonicalName());
156                 body = t.toString();//new String(t.getContents());
157                 body += t.get().toString();
158             }
159             log.log(Level.INFO, "outbound messsage for " + url + " has a body of type " + msg.getBody().getClass().getName());
160         } catch (Exception ex) {
161             log.log(Level.WARN, "fgsms Agent for JbossESB, outbound message for " + url + " has a non-text body", ex);
162 
163         }
164 
165         IMessageProcessor mp = MessageProcessor.getSingletonObject();
166         if (msgsize == null) {
167             mp.processMessageOutput(id, body, body.length(), fault, Long.valueOf(dod),new HashMap());
168         } else {
169             mp.processMessageOutput(id, body, msgsize, fault, Long.valueOf(dod),new HashMap());
170         }
171         return msg;
172     }

231                     log.log(Level.INFO, "fgsms Agent for JbossESB, inbound message is internal esb traffic " + url);

233                 }

312             action = "urn:undeterminable";
