133     public GetMyScheduledReportsResponseMsg getMyScheduledReports(
134             @WebParam(name = "request", targetNamespace = "urn:org:miloss:fgsms:services:interfaces:automatedReportingService") GetMyScheduledReportsRequestMsg request)
135             throws AccessDeniedException, ServiceUnavailableException {
136         String currentUser = UserIdentityUtil.getFirstIdentityToString(ctx);
137         if (request == null) {
138             AuditLogger.logItem(this.getClass().getCanonicalName(), "getMyScheduledReports", currentUser, "", "not specified", ctx.getMessageContext());
139             throw new IllegalArgumentException("request is null");
140         }
141         Utility.validateClassification(request.getClassification());
142 
143         AuditLogger.logItem(this.getClass().getCanonicalName(), "getMyScheduledReports", currentUser, "", (request.getClassification()), ctx.getMessageContext());
144         //log.log(Level.INFO, name + "exportDataToHTML" + currentUser);
145         //no authorization is necessary here
146 
147         if (request.getRecordlimit() == 0) {
148             request.setRecordlimit(100);
149         }
150         //when reading the database by sure to use the column to override the lastupdated field
151         GetMyScheduledReportsResponseMsg ret = new GetMyScheduledReportsResponseMsg();
152         Connection con = null;
153         PreparedStatement cmd = null;
154         ResultSet rs = null;
155         try {
156             ret.setClassification(getCurrentOperatingClassificationLevel());
157             JAXBContext GetARSSerializationContext = Utility.getARSSerializationContext();
158             Unmarshaller u = GetARSSerializationContext.createUnmarshaller();
159             con = Utility.getPerformanceDBConnection();
160             cmd = con.prepareStatement("select * from arsjobs where owninguser=?;");
161             cmd.setString(1, currentUser);
162             rs = cmd.executeQuery();
163             while (rs.next()) {
164                 ExistingReportDefitions ed = new ExistingReportDefitions();
165                 byte[] s = rs.getBytes("reportdef");
166                 ByteArrayInputStream bss = new ByteArrayInputStream(s);
167                 XMLInputFactory xf = XMLInputFactory.newInstance();
168                 XMLStreamReader r = xf.createXMLStreamReader(bss);
169                 JAXBElement<ReportDefinition> foo = (JAXBElement<ReportDefinition>) u.unmarshal(r, ReportDefinition.class);
170                 if (foo != null && foo.getValue() != null) {
171                     ed.setJob(foo.getValue());
172                 }
173                 ed.getJob().setLastRanAt(ConvertToXmlGreg(rs.getLong("lastranat")));
174                 ret.getCompletedJobs().add(ed);
175             }
176         } catch (Exception ex) {
177             log.log(Level.ERROR, null, ex);
178         } finally {
179             DBUtils.safeClose(rs);
180             DBUtils.safeClose(cmd);
181             DBUtils.safeClose(con);
182         }
183         con = null;
184 
185         try {
186             con = Utility.getPerformanceDBConnection();
187 
188             //check for completed jobs
189             for (int i = 0; i < ret.getCompletedJobs().size(); i++) {
190 
191                 cmd = con.prepareStatement("select * from arsreports where jobid=? limit  ? offset ?");
192                 cmd.setString(1, ret.getCompletedJobs().get(i).getJob().getJobId());
193                 cmd.setInt(2, request.getRecordlimit());
194                 cmd.setInt(3, request.getOffset());
195                 rs = cmd.executeQuery();
196                 while (rs.next()) {
197                     CompletedJobs cj = new CompletedJobs();
198                     cj.setReportId(rs.getString("reportid"));
199                     cj.setTimestamp(ConvertToXmlGreg(rs.getLong("datetime")));
200                     ret.getCompletedJobs().get(i).getReports().add(cj);
201                 }
202 
203                 rs.close();
204                 cmd.close();
205             }
206             return ret;
207         } catch (Exception ex) {
208             log.log(Level.ERROR, null, ex);
209         } finally {
210             DBUtils.safeClose(rs);
211             DBUtils.safeClose(cmd);
212             DBUtils.safeClose(con);
213         }
214         ServiceUnavailableException f = new ServiceUnavailableException("", null);
215         f.getFaultInfo().setCode(ServiceUnavailableFaultCodes.DATA_BASE_UNAVAILABLE);
216         throw f;
217

220     private ReportDefinition loadReportDef(String job) throws Exception {
221         Connection con = null;
222         PreparedStatement cmd = null;
223         ResultSet rs = null;
224         try {
225             ReportDefinition ret = null;
226             JAXBContext GetARSSerializationContext = Utility.getARSSerializationContext();
227             Unmarshaller u = GetARSSerializationContext.createUnmarshaller();
228             con = Utility.getPerformanceDBConnection();
229             cmd = con.prepareStatement("select * from arsjobs where jobid=?;");
230             cmd.setString(1, job);
231             rs = cmd.executeQuery();
232             if (rs.next()) {
233 
234                 byte[] s = rs.getBytes("reportdef");
235                 ByteArrayInputStream bss = new ByteArrayInputStream(s);
236                 XMLInputFactory xf = XMLInputFactory.newInstance();
237                 XMLStreamReader r = xf.createXMLStreamReader(bss);
238                 JAXBElement<ReportDefinition> foo = (JAXBElement<ReportDefinition>) u.unmarshal(r, ReportDefinition.class);
239                 if (foo != null && foo.getValue() != null) {
240                     ret = foo.getValue();
241                 }
242 
243             }
244 
245             return ret;
246         } catch (Exception ex) {
247             log.log(Level.WARN, "loadReportDef error", ex);
248         } finally {
249             DBUtils.safeClose(rs);
250             DBUtils.safeClose(cmd);
251             DBUtils.safeClose(con);
252         }
253         throw new IllegalArgumentException("job not found");

267     private GetGlobalPolicyResponseMsg GetGlobalPolicyFromDB() throws org.miloss.fgsms.services.interfaces.policyconfiguration.ServiceUnavailableException, ServiceUnavailableException {
268 
269         Connection con = Utility.getConfigurationDBConnection();
270         PreparedStatement comm = null;
271         ResultSet results = null;
272         boolean foundPolicy = false;
273         GetGlobalPolicyResponseMsg response = new GetGlobalPolicyResponseMsg();
274         try {
275             response.setPolicy(new GlobalPolicy());
276             //we don't care about the request in this case
277 
278             comm = con.prepareStatement("Select * from GlobalPolicies;");
279 
280             /////////////////////////////////////////////
281             //get the global policy for data retension
282             /////////////////////////////////////////////
283             results = comm.executeQuery();
284             //       int count = 0;
285 
286             if (results.next()) {
287                 response.getPolicy().setPolicyRefreshRate(df.newDuration(results.getLong("PolicyRefreshRate")));
288                 response.getPolicy().setRecordedMessageCap(results.getInt("RecordedMessageCap"));
289                 SecurityWrapper wrap = new SecurityWrapper(ClassificationType.fromValue(results.getString("classification")),
290                         results.getString("caveat"));
291                 response.setClassification(wrap);
292                 response.getPolicy().setAgentsEnabled(results.getBoolean("agentsenable"));
293                 foundPolicy = true;
294             }
295         } catch (Exception ex) {
296             log.log(Level.ERROR, "error getting global policy", ex);
297             ServiceUnavailableException f = new ServiceUnavailableException("", null);
298             f.getFaultInfo().setCode(ServiceUnavailableFaultCodes.DATA_BASE_UNAVAILABLE);
299             throw f;
300         } finally {
301             DBUtils.safeClose(results);
302             DBUtils.safeClose(comm);
303             DBUtils.safeClose(con);
304         }
305 
306         try {
307             if (!foundPolicy) {
308                 try {
309                     comm = con.prepareStatement("INSERT INTO GlobalPolicies (PolicyRefreshRate, RecordedMessageCap, classification, agentsenable, caveat) "
310                             + " VALUES (?, ?, ?, true, ?);");
311                     comm.setLong(1, 30 * 60 * 100);
312                     comm.setLong(2, 1024000);
313                     comm.setString(3, "U");
314                     comm.setString(4, "");
315                     comm.execute();
316                     response.getPolicy().setRecordedMessageCap(1024000);
317                     response.getPolicy().setClassification(new SecurityWrapper(ClassificationType.U, ""));
318                     response.getPolicy().setAgentsEnabled(true);
319                 } catch (SQLException ex) {
320                     log.log(Level.ERROR, "error setting global policy", ex);
321                 }
322             }
323             KeyNameValueEnc d = DBSettingsLoader.GetPropertiesFromDB(true, "UddiPublisher", "Interval");
324             if (d != null && d.getKeyNameValue() != null) {
325                 try {
326                     response.getPolicy().setUDDIPublishRate(df.newDuration(Long.parseLong(d.getKeyNameValue().getPropertyValue())));
327                 } catch (Exception ex) {
328                     response.getPolicy().setUDDIPublishRate(df.newDuration(30 * 60 * 100));
329                 }
330             }
331             //response.setClassification(getCurrentOperatingClassificationLevel());
332             return response;
333         } catch (Exception ex) {
334             log.log(Level.ERROR, "error getting global policy", ex);
335             ServiceUnavailableException f = new ServiceUnavailableException("", null);
336             f.getFaultInfo().setCode(ServiceUnavailableFaultCodes.DATA_BASE_UNAVAILABLE);
337             throw f;
338         } finally {
339             DBUtils.safeClose(comm);
340             DBUtils.safeClose(con);
341         }
342 
343     }

357     public AddOrUpdateScheduledReportResponseMsg addOrUpdateScheduledReport(
358             @WebParam(name = "request", targetNamespace = "urn:org:miloss:fgsms:services:interfaces:automatedReportingService") AddOrUpdateScheduledReportRequestMsg request)
359             throws AccessDeniedException, ServiceUnavailableException {
360         String currentUser = UserIdentityUtil.getFirstIdentityToString(ctx);
361         if (request == null) {
362             AuditLogger.logItem(this.getClass().getCanonicalName(), "addOrUpdateScheduledReport", currentUser, "", "not specified", ctx.getMessageContext());
363             throw new IllegalArgumentException("request is null");
364         }
365         Utility.validateClassification(request.getClassification());
366         if (request.getJobs().isEmpty()) {
367             throw new IllegalArgumentException("at least one report must be specified for updating");
368         }
369         AuditLogger.logItem(this.getClass().getCanonicalName(), "addOrUpdateScheduledReport", currentUser, "", (request.getClassification()), ctx.getMessageContext());
370 
371         //validate request
372         //loop through all jobs, add job ids if not present, authorize the request
373         validateRequest(request, currentUser, ctx);
374 
375         //     if (UserIdentityUtil.hasGlobalAdministratorRole(currentUser, "addOrUpdateScheduledReport", request.getClassification(), ctx))
376         //TODO admins?
377         Connection con = Utility.getPerformanceDBConnection();
378         for (int i = 0; i < request.getJobs().size(); i++) {
379             PreparedStatement cmd = null;
380             try {
381 
382                 StringWriter sw = new StringWriter();
383 
384                 if (!Utility.stringIsNullOrEmpty(request.getJobs().get(i).getJobId())) {
385                     cmd = con.prepareStatement("UPDATE arsjobs SET  reportdef=?, hasextrapermissions=?, enabled=? WHERE jobid=?;");
386                     //cmd.setBytes(1, bits);
387                     cmd.setBoolean(2, !request.getJobs().get(i).getAdditionalReaders().isEmpty());
388                     cmd.setBoolean(3, request.getJobs().get(i).isEnabled());
389                     cmd.setString(4, request.getJobs().get(i).getJobId());
390                 } else {
391                     request.getJobs().get(i).setJobId(UUID.randomUUID().toString());
392                     cmd = con.prepareStatement("INSERT INTO arsjobs (reportdef, hasextrapermissions,  enabled, lastranat, jobid, owninguser) values (?,?,?,?,?,?);");
393 
394                     cmd.setBoolean(2, !request.getJobs().get(i).getAdditionalReaders().isEmpty());
395 
396                     cmd.setBoolean(3, request.getJobs().get(i).isEnabled());
397                     cmd.setLong(4, 0);
398                     cmd.setString(5, request.getJobs().get(i).getJobId());
399                     cmd.setString(6, currentUser);
400                 }
401                 Utility.getARSSerializationContext().createMarshaller().marshal(request.getJobs().get(i), sw);
402                 byte[] bits = sw.toString().getBytes(Constants.CHARSET);
403                 cmd.setBytes(1, bits);
404                 cmd.executeUpdate();
405                 cmd.close();
406             } catch (Exception ex) {
407                 log.log(Level.ERROR, null, ex);
408             } finally {
409                 DBUtils.safeClose(cmd);
410             }
411         }
412         if (con != null) {
413             DBUtils.safeClose(con);
414         }
415         AddOrUpdateScheduledReportResponseMsg r = new AddOrUpdateScheduledReportResponseMsg();
416         r.setClassification(getCurrentOperatingClassificationLevel());
417         r.getJobs().addAll(request.getJobs());
418         r.setSuccess(true);
419         r.setMessage(OK);
420         return r;

758     private static void validReportDefinition(ReportDefinition get) {
759         if (get == null) {
760             throw new IllegalArgumentException("The report def is empty");
761         }
762         if (get.getSchedule() == null) {
763             throw new IllegalArgumentException("The schedule is empty");
764         }
765         if (get.getSchedule().getTriggers().isEmpty()) {
766             throw new IllegalArgumentException("The report def has no triggers defined.");
767         }
768 
769         for (int i = 0; i < get.getSchedule().getTriggers().size(); i++) {
770             if (get.getSchedule().getTriggers().get(i).getStartingAt() == null) {
771                 throw new IllegalArgumentException("The report def has invalid triggers.");
772             }
773 
774             if (get.getSchedule().getTriggers().get(i) instanceof DailySchedule) {
775                 DailySchedule ds = (DailySchedule) get.getSchedule().getTriggers().get(i);
776                 if (ds.getReoccurs() == null || ds.getReoccurs().intValue() < 1) {
777                     throw new IllegalArgumentException("The report def has invalid value for reoccuring.");
778                 }
779             } else if (get.getSchedule().getTriggers().get(i) instanceof WeeklySchedule) {
780                 WeeklySchedule ds = (WeeklySchedule) get.getSchedule().getTriggers().get(i);
781                 if (ds.getDayOfTheWeekIs().isEmpty()) {
782                     throw new IllegalArgumentException("The report def has invalid weekly schedule for day of the weeks.");
783                 }
784                 if (ds.getReoccurs() == null || ds.getReoccurs().intValue() < 1) {
785                     throw new IllegalArgumentException("The report def has invalid value for reoccuring.");
786                 }
787             } else if (get.getSchedule().getTriggers().get(i) instanceof MonthlySchedule) {
788                 MonthlySchedule ds = (MonthlySchedule) get.getSchedule().getTriggers().get(i);
789                 if (ds.getDayOfTheMonthIs().isEmpty()) {
790                     throw new IllegalArgumentException("The report def has invalid weekly schedule for day of the month.");
791                 }
792                 if (ds.getMonthNameIs().isEmpty()) {
793                     throw new IllegalArgumentException("The report def has invalid weekly schedule for month.");
794                 }
795             } else if (get.getSchedule().getTriggers().get(i) instanceof OneTimeSchedule) {
796                 //TODO
797             } else if (get.getSchedule().getTriggers().get(i) instanceof ImmediateSchedule) {
798                 //TODO
799             } else {
800                 throw new IllegalArgumentException("The report def has a trigger that is not intrepretable.");
801             }
802         }
803 
804         if (get.getExportCSVDataRequestMsg() != null) {
805             Utility.validateClassification(get.getExportCSVDataRequestMsg().getClassification());
806         }
807         if (get.getExportDataRequestMsg() != null) {
808             Utility.validateClassification(get.getExportDataRequestMsg().getClassification());
809         }
810     }

812     private static void validateRequest(AddOrUpdateScheduledReportRequestMsg request, String currentUser, WebServiceContext ctx) throws AccessDeniedException {
813         for (int i = 0; i < request.getJobs().size(); i++) {
814             request.getJobs().get(i).setOwner(currentUser);
815             if (Utility.stringIsNullOrEmpty(request.getJobs().get(i).getJobId())) {
816                 //its a new job
817                 //request.getJobs().get(i).setJobId(UUID.randomUUID().toString());
818             } else {
819                 //existing job
820                 //prevent users from overriding existing reports that are owned by others
821                 if (!IsReportJobOwner(currentUser, request.getJobs().get(i).getJobId())) {
822                     AccessDeniedException f = new AccessDeniedException("the report job " + request.getJobs().get(i).getJobId() + " is not owned by you", null);
823                     throw f;
824                 }
825             }
826             if (request.getJobs().get(i).getExportCSVDataRequestMsg() == null && request.getJobs().get(i).getExportDataRequestMsg() == null) {
827                 throw new IllegalArgumentException("one of ExportData or ExportCSV must be specified");
828             }
829             if (request.getJobs().get(i).getExportCSVDataRequestMsg() != null && request.getJobs().get(i).getExportDataRequestMsg() != null) {
830                 throw new IllegalArgumentException("both ExportData and ExportCSV cannot be specified on the same report definition");
831             }
832 
833             validReportDefinition(request.getJobs().get(i));
834             validateReportAlerts(request.getJobs().get(i));
835             if (request.getJobs().get(i).getExportCSVDataRequestMsg() != null) {
836                 if (request.getJobs().get(i).getExportCSVDataRequestMsg().getExportType() != ExportRecordsEnum.AUDIT_LOGS
837                         && request.getJobs().get(i).getExportCSVDataRequestMsg().getURLs().isEmpty()) {
838                     throw new IllegalArgumentException("ExportCSV requires at least one URL when not requesting audit logs");
839                 }
840                 for (int k = 0; k < request.getJobs().get(i).getExportCSVDataRequestMsg().getURLs().size(); k++) {
841                     if (request.getJobs().get(i).getExportCSVDataRequestMsg().getExportType() == ExportRecordsEnum.TRANSACTIONS) {
842                         UserIdentityUtil.assertAuditAccess(request.getJobs().get(i).getExportCSVDataRequestMsg().getURLs().get(k), currentUser, "addOrUpdateScheduledReport", request.getClassification(), ctx);
843                     } else {
844                         UserIdentityUtil.assertReadAccess(request.getJobs().get(i).getExportCSVDataRequestMsg().getURLs().get(k), currentUser, "addOrUpdateScheduledReport", request.getClassification(), ctx);
845                     }
846                 }
847                 assertNotNull(request.getJobs().get(i).getExportCSVDataRequestMsg().getExportType());
848                 assertNotNull(request.getJobs().get(i).getExportCSVDataRequestMsg().getRange());
849                 assertNotNull(request.getJobs().get(i).getExportCSVDataRequestMsg().getRange().getEnd());
850                 assertNotNull(request.getJobs().get(i).getExportCSVDataRequestMsg().getRange().getStart());
851                 validateRange(request.getJobs().get(i).getExportCSVDataRequestMsg().getRange());
852                 if (request.getJobs().get(i).getExportCSVDataRequestMsg().getExportType() == ExportRecordsEnum.AUDIT_LOGS) {
853                     UserIdentityUtil.assertAuditRole(currentUser, "ValidReportDefinition", request.getClassification(), ctx);
854                 }
855             }
856             if (request.getJobs().get(i).getExportDataRequestMsg() != null) {
857                 for (int k = 0; k < request.getJobs().get(i).getExportDataRequestMsg().getURLs().size(); k++) {
858                     UserIdentityUtil.assertReadAccess(request.getJobs().get(i).getExportDataRequestMsg().getURLs().get(k), currentUser, "addOrUpdateScheduledReport", request.getClassification(), ctx);
859                 }
860 
861                 assertNotNull(request.getJobs().get(i).getExportDataRequestMsg().getReportTypes());
862                 assertNotNull(request.getJobs().get(i).getExportDataRequestMsg().getReportTypes().getReportTypeContainer());
863                 assertFalse(request.getJobs().get(i).getExportDataRequestMsg().getReportTypes().getReportTypeContainer().isEmpty());
864                 assertNotNull(request.getJobs().get(i).getExportDataRequestMsg().getRange());
865                 assertNotNull(request.getJobs().get(i).getExportDataRequestMsg().getRange().getEnd());
866                 assertNotNull(request.getJobs().get(i).getExportDataRequestMsg().getRange().getStart());
867                 validateRange(request.getJobs().get(i).getExportDataRequestMsg().getRange());
868 
869             }
870         }
